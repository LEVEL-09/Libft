void    *ft_memmove(void *dest, const void *src, size_t n)
{
        unsigned char   *ptr;

        if (!dest && !src)
                return (NULL);
        ptr = (unsigned char *)dest;
        if (src < dest)
        {
                while (n-- > 0)
                {
                        *(ptr + n) = *((unsigned char *)src + n);
                }
        }
        else
                ft_memcpy (dest, src, n);
        return ((unsigned char *)dest);
}
